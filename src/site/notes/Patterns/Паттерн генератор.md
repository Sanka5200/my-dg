```class AbstractPage:
    def backgroundColor(self):
        return "undefined"
    def textColor(self):
        return "undefined"
    def printPage(self):
        assert self.backgroundColor() != "undefined" and self.textColor() != "undefined"
        backgroundColor = self.backgroundColor()
        textColor = self.textColor()
        print(f"""<!DOCTYPE html>
<html lang="ru">
<head>
    <meta charset="UTF-8">
    <title>Досье страницы</title>
    <style>
        body {{
            background-color: {backgroundColor};
            color: {textColor};
            font-family: Arial, sans-serif;
            text-align: center;
            height: 100vh;
            display: flex;
            justify-content: center;
            align-items: center;
        }}
    </style>
</head>
<body>
    <h1>я чёрно-белая страница</h1>
</body>
</html>""")

class WhitePage(AbstractPage):
    def backgroundColor(self):
        return "white"
    def textColor(self):
        return "black"
        
class BlackPage(AbstractPage):
    def backgroundColor(self):
        return "black"
    def textColor(self):
        return "white"


p1 = WhitePage()
p1.printPage()
print("\n\n\n")
p2 = BlackPage()
p2.printPage()
